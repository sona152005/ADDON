package Collection.map;
import java.util.HashMap;

public class HashMapClass {
	public static void main(String[] args) {
		HashMap <String,Integer> student = new HashMap <>();
		student.put("sona",90);
		student.put("anujha",95);
		student.put("maha",96);
		
		int total = 0;
        for (int mark : student.values()) {
            total += mark;
        }
        double average = 0;
        if (!student.isEmpty()) {
            average = (double)total / student.size();
        }

       
        System.out.println("Student Marks:");
        for(String mark: student.keySet())
            System.out.println(mark+ ": " + student.get(mark));
        
 System.out.println("average"+average);
			
	}
}

output:
Student Marks:
anujha: 95
sona: 90
maha: 96
average93.66666666666667
